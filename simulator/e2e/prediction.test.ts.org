import { test, expect } from '@playwright/test';

// 로그인 헬퍼 함수
async function login(page: any) {
    await page.goto('/');

    const loginButton = page.getByRole('button', { name: /login/i }).first();
    if (await loginButton.isVisible({ timeout: 3000 })) {
        await loginButton.click();
        await page.fill('input[type="email"]', 'test1@mail.com');
        await page.fill('input[type="password"]', '123456');
        await page.locator('button[type="submit"]').click();
        await page.waitForTimeout(3000); // 로그인 완료 대기
    }
}

test.describe('예측 제출 테스트 (Prediction Submission Tests)', () => {
    test.beforeEach(async ({ page }) => {
        await login(page);
    });

    test('TC-PRED-01: 기본 UP 예측 제출 (Submit UP Prediction)', async ({ page }) => {
        // 메인 페이지 이동
        await page.goto('/');

        // UP 버튼 선택
        const upButton = page.getByRole('button', { name: /up|bull/i }).first();
        await upButton.click();

        // 목표 퍼센트 선택 (0.5%)
        const targetButton = page.locator('button:has-text("0.5%")').or(
            page.locator('[data-target="0.5"]')
        ).first();
        if (await targetButton.isVisible({ timeout: 2000 })) {
            await targetButton.click();
        }

        // 베팅 금액 입력
        const betInput = page.locator('input[type="number"]').or(
            page.locator('input[placeholder*="amount"]')
        ).first();
        await betInput.fill('50');

        // 예측 제출
        const submitButton = page.getByRole('button', { name: /place|submit|predict/i }).first();
        await submitButton.click();

        // 성공 메시지 확인
        await expect(
            page.locator('text=/success|submitted|placed/i').first()
        ).toBeVisible({ timeout: 10000 });
    });

        ).first();

            await targetButton.click();
        }

        // 베팅 금액 입력
